#集群框架文档
## 一. 框架介绍
- 有些服务端程序在设计时，会基于主/副节点的集群方式来进行设计，即：从所有对等的节点中选举出一个主节点，而后由主节点进行逻辑的统一协调。
- 集群框架就是为简化这一过程而设计的。使用集群框架的程序，只需要考虑master需要做什么业务，slave需要做什么业务即可，以及对一些集群事件进行处理即可。而不必关心具体的选主、心跳保持等实现细节。
###下边这段代码是一个集群节点启动的配置：

    public class Node1 {
        public static void main(String[] args) throws Exception {
    
            ClusterConfigBuilder configBuilder = new ClusterConfigBuilder();
            configBuilder.setLocalUID("node#1")
                    .setMasterPort(61618)
                    .setMasterAcceptorThreadCount(3)
                    .setMasterWorkerThreadCount(10)
                    .setClusterName("seo_engine")
                    .setMasterExposureIpPrefix("192.");
    
            // 注册集群的事件handler
            configBuilder.addHandler(new BaseHandler() {
                @Override
                public boolean onBecameMaster(MasterInfo masterInfo) {
                    return false;
                }
    
                @Override
                public boolean onBecameFollower() {
                    return false;
                }
    
                @Override
                public boolean onAboutToReleaseMaster() {
                    return false;
                }
    
                @Override
                public boolean onFollowerEnterIn(Follower follower) {
                    return false;
                }
    
                @Override
                public boolean onUserMsgFromMaster(UserDataBody data) {
                    return false;
                }
    
                @Override
                public String getHandlerName() {
                    return "DemoHandler";
                }
    
                @Override
                public boolean onUserMsgFromFollower(UserDataBody data, Follower follower) {
                    return false;
                }
            });
    
            ClusterConfig clusterConfig = configBuilder.build();
    
            // 初始化一个选举器（目前只实现了zookeeper选举器）
            ZkElectorConfig zkElectorConfig = new ZkElectorConfig("127.0.0.1:2181", 3000, "/yp_seo_engine");
            ClusterBootstrap node1 = new ClusterBootstrap(clusterConfig, new ZkElector(clusterConfig, zkElectorConfig));
    
            // 启动集群节点
            node1.startup();
    
            Thread.sleep(999999999L);
        }
    }

### 我们也可以使用以下代码，在一个进程中启动多个集群节点：

    public class TestClusterInOneProcessor {
        public static void main(String[] args) throws Exception {
            ClusterConfigBuilder configBuilder = new ClusterConfigBuilder();
            configBuilder.setClusterNetworkCardName("vboxnet0")
                    .setLocalUID("node#1")
                    .setMasterPort(61618)
                    .setMasterAcceptorThreadCount(3)
                    .setMasterWorkerThreadCount(10)
                    .setClusterName("seo_engine")
                    .setMasterExposureIpPrefix("192.");
    
            ClusterConfig config1 = configBuilder.build();
    
            configBuilder.setLocalUID("node#2").setMasterPort(61619);
            ClusterConfig config2 = configBuilder.build();
    
            configBuilder.setLocalUID("node#3").setMasterPort(61620);
            ClusterConfig config3 = configBuilder.build();
    
            configBuilder.setLocalUID("node#4").setMasterPort(61621);
            ClusterConfig config4 = configBuilder.build();
    
            ZkElectorConfig zkElectorConfig = new ZkElectorConfig("127.0.0.1:2181", 3000, "/yp_seo_engine");
            ClusterBootstrap cluster1 = new ClusterBootstrap(config1, new ZkElector(config1, zkElectorConfig));
            ClusterBootstrap cluster2 = new ClusterBootstrap(config2, new ZkElector(config2, zkElectorConfig));
            ClusterBootstrap cluster3 = new ClusterBootstrap(config3, new ZkElector(config3, zkElectorConfig));
            ClusterBootstrap cluster4 = new ClusterBootstrap(config4, new ZkElector(config4, zkElectorConfig));
    
            cluster1.startup();
            cluster2.startup();
            cluster3.startup();
            cluster4.startup();
    
            Thread.sleep(999999999L);
        }
    }
